  // 以前
  Cnd cnd = Cnd.where("age", "<", 25);
  if (!Strings.isBlank(city)) {
      cnd.and("city", "=", city);
  }

  Cnd cnd = Cnd.where("age", "<", 25);
  cnd.andEX("city", "=", city);
