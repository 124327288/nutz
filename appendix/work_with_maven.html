将Nutz加入你的POM

<dependency>
	<groupId>org.nutz</groupId>
	<artifactId>nutz</artifactId>
	<version>1.r.59</version>
	<exclusions>
		<exclusion>
			<artifactId>servlet-api</artifactId>
			<groupId>javax.servlet</groupId>
		</exclusion>
	</exclusions>
</dependency>
